Variant #1:


const array = [1, 2, 3, 4, 5, 6, 7];

function removeElement1(array, num) {
  const resultArray = [];
  for (let i of array) {
    if (i === num) { 
      continue;
    }
    else { 
      resultArray.push(i);
    }
  }
  return resultArray;
}

const array = [1, 2, 3, 4, 5, 6, 7];

function removeElement2(array, num) {
  tempArray = [...array];
  tempArray.splice(array.indexOf(num), 1);
  return tempArray;
}
